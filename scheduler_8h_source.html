130 void threadFn() {
131 if (name.has_value()) {
132 // loguru::set_thread_name(name->c_str());
133 }
134 std::unique_lock lock(scheduleMutex);
135 while (!quitting) {
136 if (schedule.empty()) {
137 scheduleCV.wait(lock, [&] { return newEventAdded; });
138 newEventAdded = false;
139 } else {
140 auto now = Clock::now();
141 schedule_t event = schedule.top();
142 if (toRemove.find(event.id) != toRemove.end()) {
143 // if we should remove this id then remove the event and don't reschedule
144 schedule.pop();
145 toRemove.erase(event.id);
146 } else if (event.nextSendTime <= now) {
147 // execute the event and reschedule it
148 schedule.pop();
149 event.fn();
150 event.nextSendTime += event.period;
151 schedule.push(event);
152 } else {
153 // wait until the event should be executed
154 scheduleCV.wait_until(lock, event.nextSendTime,
155 [&] { return newEventAdded; });
156 newEventAdded = false;
157 }
158 }
159 }
